project( MetaImageImporter )

cmake_minimum_required( VERSION 2.4 )
if( COMMAND CMAKE_POLICY )
  cmake_policy( SET CMP0003 NEW )
endif( COMMAND CMAKE_POLICY )

if( TubeTK_USE_QT )

  # Find QT
  find_package( Qt4 REQUIRED )
  include( ${QT_USE_FILE} )

  include_directories(
    ${QT_INCLUDE_DIR}
    ${QT_QTGUI_INCLUDE_DIR}
    ${QT_QTCORE_INCLUDE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR} )

  set( MetaImageImporter_SRCS
    main.cxx
    MetaImageImporterUI.h
    MetaImageImporterUI.cxx )

  set( MetaImageImporter_MOC_SRCS
    MetaImageImporterUI.h )
  set( MetaImageImporter_UI_SRCS MetaImageImporterUI.ui )

  QT4_WRAP_CPP( MetaImageImporter_SRCS ${MetaImageImporter_MOC_SRCS} )
  QT4_WRAP_UI( MetaImageImporter_UI_CXX ${MetaImageImporter_UI_SRCS} )

  option( HIDE_CONSOLE "Hide the console" OFF )
  if( HIDE_CONSOLE )
    set( CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /subsystem:windows" )
    link_libraries( ${QT_QTMAIN_LIBRARY} )
  endif( HIDE_CONSOLE )

  add_executable( MetaImageImporter
    ${MetaImageImporter_SRCS} ${MetaImageImporter_UI_CXX} )

  link_libraries(
    ${QT_QTCORE_LIBRARY}
    ${QT_QTGUI_LIBRARY} )

  target_link_libraries( MetaImageImporter
    ${QT_QTCORE_LIBRARY}
    ${QT_QTGUI_LIBRARY}
    ${QT_QTDESIGNER_LIBRARY} )

  # Add the install rule
  install( FILES DESTINATION bin COMPONENT RUNTIME )

  # Installation file
  install(
    TARGETS MetaImageImporter
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    RUNTIME DESTINATION bin
    COMPONENT RUNTIME )

endif( TubeTK_USE_QT )

add_executable( MetaImageImporterCLI MetaImageImporterCLI.cxx )
